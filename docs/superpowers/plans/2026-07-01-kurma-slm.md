# Kurma SLM Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an agent workflow governance harness — blueprint-driven, evaluation engine with deterministic rules + LLM judge fallback, live-mode write gating, and HITL decision capture.

**Architecture:** Each module is one file with one responsibility. A 9B SLM implements one file at a time. Files talk to each other through typed dataclasses defined in `src/models.py` (shared first). No file exceeds ~150 lines.

**Tech Stack:** Python 3.11+, dataclasses, pydantic v2, PyYAML, pytest, ruff

## Global Constraints

- Python 3.11+ only — use `dataclasses.dataclass`, `from __future__ import annotations`
- All models in `src/models.py` — no model definitions in other files
- All output files go under `kurma_slm/` — never modify files outside that folder
- Tests live in `kurma_slm/tests/` mirroring `src/` names (e.g. `src/blueprint.py` → `tests/test_blueprint.py`)
- No external LLM calls in unit tests — mock at the boundary
- Every file must pass `ruff check` with zero issues
- Commit after every task

# kurma_slm/src/blueprint.py
from __future__ import annotations
import json
from pathlib import Path
from typing import Any
import yaml
from src.models import Blueprint, BlueprintAction

_REQUIRED = {"id", "name", "version", "workflow_id", "mode", "steps"}

def load_blueprint(path: str | Path) -> Blueprint:
    path = Path(path)
    text = path.read_text(encoding="utf-8")
    data: dict[str, Any] = yaml.safe_load(text) if path.suffix in (".yaml", ".yml") else json.loads(text)

    missing = _REQUIRED - data.keys()
    if missing:
        raise ValueError(f"Blueprint missing required fields: {', '.join(sorted(missing))}")

    steps = [
        BlueprintAction(
            id=s["id"],
            name=s["name"],
            kind=s["kind"],
            mandatory=s.get("mandatory", s["kind"] == "required"),
            conditions=s.get("conditions", []),
            allowed_tools=s.get("allowed_tools", []),
        )
        for s in data["steps"]
    ]
    return Blueprint(
        id=data["id"],
        name=data["name"],
        version=data["version"],
        workflow_id=data["workflow_id"],
        mode=data["mode"],
        steps=steps,
    )
```

# kurma_slm/src/versioning.py
from __future__ import annotations
from src.models import Blueprint

class BlueprintVersionStore:
    def __init__(self) -> None:
        self._store: dict[str, dict[str, Blueprint]] = {}

    def save(self, bp: Blueprint) -> None:
        self._store.setdefault(bp.workflow_id, {})[bp.version] = bp

    def get(self, workflow_id: str, version: str) -> Blueprint:
        try:
            return self._store[workflow_id][version]
        except KeyError:
            raise KeyError(f"Blueprint {workflow_id!r} version {version!r} not found")

    def list(self, workflow_id: str) -> list[str]:
        return sorted(self._store.get(workflow_id, {}).keys())

    def rollback(self, workflow_id: str, version: str) -> Blueprint:
        return self.get(workflow_id, version)
```

# kurma_slm/src/harness.py
from __future__ import annotations
import uuid
from typing import Any, Callable
from src.models import TraceEntry

class AgentHarness:
    def __init__(self, run_id: str | None = None) -> None:
        self._run_id = run_id or str(uuid.uuid4())
        self._tools: dict[str, dict[str, Any]] = {}
        self._trace: list[TraceEntry] = []

    def register(self, name: str, kind: str, fn: Callable[[dict[str, Any]], Any]) -> None:
        self._tools[name] = {"kind": kind, "fn": fn}

    def call(self, tool: str, args: dict[str, Any]) -> Any:
        if tool not in self._tools:
            raise KeyError(f"Tool {tool!r} not registered")
        entry = self._tools[tool]
        output = entry["fn"](args)
        self._trace.append(TraceEntry(
            run_id=self._run_id, tool=tool,
            kind=entry["kind"], args=args, output=output,
        ))
        return output

    def get_trace(self) -> list[TraceEntry]:
        return list(self._trace)
```

# kurma_slm/src/validator.py
from __future__ import annotations
from typing import Any, Type
from pydantic import BaseModel, ValidationError

class PreFlightValidator:
    def __init__(self) -> None:
        self._schemas: dict[str, Type[BaseModel]] = {}

    def register_schema(self, tool: str, schema: Type[BaseModel]) -> None:
        self._schemas[tool] = schema

    def validate(self, tool: str, args: dict[str, Any]) -> tuple[bool, str | None]:
        if tool not in self._schemas:
            return True, None
        try:
            self._schemas[tool](**args)
            return True, None
        except ValidationError as e:
            return False, str(e)
```

# kurma_slm/src/evaluator.py
from __future__ import annotations
from src.models import Blueprint, TraceEntry, DeviationResult

class EvaluationEngine:
    def evaluate(self, blueprint: Blueprint, trace: list[TraceEntry]) -> list[DeviationResult]:
        deviations: list[DeviationResult] = []
        tools_called = {e.tool for e in trace}
        all_allowed = {t for step in blueprint.steps for t in step.allowed_tools}

        for step in blueprint.steps:
            called = tools_called & set(step.allowed_tools)
            if step.kind == "required" and not called:
                deviations.append(DeviationResult(
                    rule="required_missing",
                    tool=", ".join(step.allowed_tools),
                    severity="high",
                    reasoning=f"Required step '{step.name}' was not executed.",
                ))
            if step.kind == "prohibited" and called:
                deviations.append(DeviationResult(
                    rule="prohibited_action",
                    tool=", ".join(called),
                    severity="high",
                    reasoning=f"Prohibited action '{step.name}' was executed.",
                ))

        for entry in trace:
            if entry.tool not in all_allowed:
                deviations.append(DeviationResult(
                    rule="unexpected_tool",
                    tool=entry.tool,
                    severity="medium",
                    reasoning=f"Tool '{entry.tool}' is not defined in the blueprint.",
                ))

        return deviations
```

# kurma_slm/src/judge.py
from __future__ import annotations
import json
from typing import Any, Callable
from src.models import DeviationResult

_PROMPT_TEMPLATE = """\
A tool call occurred that is not covered by the blueprint.
Tool: {tool}
Context: {context}
Is this action necessary and appropriate? Reply with JSON only:
{{"severity": "low|medium|high", "reasoning": "one sentence"}}
"""

class LLMJudge:
    def assess(
        self,
        tool: str,
        context: dict[str, Any],
        call_llm: Callable[[str], str],
    ) -> DeviationResult:
        prompt = _PROMPT_TEMPLATE.format(tool=tool, context=context)
        raw = call_llm(prompt)
        try:
            data = json.loads(raw)
            severity = data.get("severity", "unknown")
            reasoning = data.get("reasoning", raw)
        except json.JSONDecodeError:
            severity = "unknown"
            reasoning = raw
        return DeviationResult(
            rule="judge_assessed",
            tool=tool,
            severity=severity,
            reasoning=reasoning,
            judge_used=True,

# kurma_slm/src/gate.py
from __future__ import annotations
import time
import uuid
from typing import Any

class WriteGate:
    def __init__(self, timeout_seconds: float = 3600.0) -> None:
        self._timeout = timeout_seconds
        self._pending: dict[str, dict[str, Any]] = {}

    def suspend(self, run_id: str, tool: str, state: dict[str, Any]) -> str:
        token = str(uuid.uuid4())
        self._pending[token] = {
            "run_id": run_id, "tool": tool,
            "state": state, "suspended_at": time.time(),
        }
        return token

    def resume(self, token: str, decision: str) -> dict[str, Any]:
        if token not in self._pending:
            raise KeyError(f"Unknown token: {token!r}")
        entry = self._pending.pop(token)
        return {"decision": decision, "tool": entry["tool"],
                "run_id": entry["run_id"], "state": entry["state"]}

    def check_timeout(self, token: str) -> bool:
        if token not in self._pending:
            return False
        elapsed = time.time() - self._pending[token]["suspended_at"]
        return elapsed > self._timeout
```

# kurma_slm/src/draft.py
from __future__ import annotations
from src.models import Variant

class DraftManager:
    def __init__(self) -> None:
        self._drafts: dict[str, list[Variant]] = {}

    def add_variant(self, variant: Variant) -> None:
        self._drafts.setdefault(variant.workflow_id, []).append(variant)

    def get_draft(self, workflow_id: str) -> list[Variant]:
        return list(self._drafts.get(workflow_id, []))

    def publish(self, workflow_id: str, selected_ids: list[str]) -> list[Variant]:
        draft = self._drafts.get(workflow_id, [])
        selected = [v for v in draft if v.id in selected_ids]
        for v in selected:
            v.status = "active"
        self._drafts[workflow_id] = [v for v in draft if v.id not in selected_ids]
        return selected
```

# kurma_slm/src/metrics.py
from __future__ import annotations
from collections import defaultdict

class MetricsCollector:
    def __init__(self) -> None:
        self._flags: list[bool] = []
        self._hitl_times: list[float] = []
        self._preflight_calls = 0
        self._preflight_blocks = 0
        self._recurrences: dict[str, int] = defaultdict(int)

    def record_flag(self, approved_by_reviewer: bool) -> None:
        self._flags.append(approved_by_reviewer)

    def record_hitl_response(self, seconds: float) -> None:
        self._hitl_times.append(seconds)

    def record_preflight_call(self) -> None:
        self._preflight_calls += 1

    def record_preflight_block(self) -> None:
        self._preflight_blocks += 1

    def record_rejected_variant_recurrence(self, signature: str) -> None:
        self._recurrences[signature] += 1

    def summary(self) -> dict:
        fp_rate = sum(self._flags) / len(self._flags) if self._flags else 0.0
        mean_hitl = sum(self._hitl_times) / len(self._hitl_times) if self._hitl_times else 0.0
        block_rate = self._preflight_blocks / self._preflight_calls if self._preflight_calls else 0.0
        return {
            "false_positive_rate": fp_rate,
            "mean_hitl_response_seconds": mean_hitl,
            "preflight_block_rate": block_rate,
            "repeat_offense_count": dict(self._recurrences),
        }
```
